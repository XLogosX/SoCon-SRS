% **************************************
% Symbol Commands:
% **************************************

% **************************************
% - checkBox
%   usage:
%      \checkBox{0} prints an empty checkbox
%      \checkBox{1} prints a filled checkbox
% **************************************
\newcommand{\checkBox}[1]{
    \ifnumequal{#1}{1}{$\boxtimes$}{$\square$}
}

% **************************************
% - auto numbering of table rows
%   usage:
%      before using the auto numbering use the \resetTabCtr to reset the numbering
%      in each row use the \rownr command to automatically print the number
% **************************************
\newcounter{magicrownumber}
\newcommand\rownr{\stepcounter{magicrownumber}\arabic{magicrownumber}}
\newcommand{\resetTabCtr}{\setcounter{magicrownumber}{0}}
